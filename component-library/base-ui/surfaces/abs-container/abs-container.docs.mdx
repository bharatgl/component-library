description: A container that opens, closes and positions another container
---

#### Example:

```jsx
// import { Container } from '@bharatgl/component-library.base-ui.surfaces.abs-container.container';
// import { Containee } from '@bharatgl/component-library.base-ui.surfaces.abs-container.containee';

class Drawer extends Component {
  state = { isOpen: this.props.open || false };
  ref = createRef();
  handleLeaveContainer() {}
  handleEnterContainer() {}
  handleContainerClick() {}
  // ...

  render() {
    return (
      <Container
        ref={this.ref}
        open={isOpen}
        onMouseLeave={this.handleLeaveContainer}
        onMouseEnter={this.handleEnterContainer}
        onClick={this.handleContainerClick}
      >
        <Containee></Containee>
      </Container>
    );
  }
}
```
